	vars: vola(0), v1(0);

	v1 = (highest(TSMFCVolatility(close,LEN),60) - lowest(TSMFCVolatility(close,LEN),60)) * 0.01;
	if v1 <> 0 then 
			vola = (TSMFCVolatility(close,LEN) - lowest(TSMFCVolatility(close,LEN),60)) / v1 
		else 
			vola=0;
	TSMFCStochVolatility = vola;
