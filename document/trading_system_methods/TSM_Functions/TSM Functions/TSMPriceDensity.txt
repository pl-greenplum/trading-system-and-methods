{ TMSPriceDensity : Price Density function
  Copyright 1999 P J Kaufman.  All rights reserved. }
  
	input:	period(numericsimple);
	vars:	vlty(0), pd(0);

	vlty = AvgTrueRange(period);
	pd = 1 - period*vlty /(period* (highest(high,period) - lowest(low,period)));
	TSMPriceDensity = pd;
