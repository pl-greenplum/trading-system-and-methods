	inputs: price(numericseries), length(numericsimple);
	vars:	force(0), sforce(0);

	force = volume * (price - price[1]);
	sforce = sforce[1] +  (2 / (length + 1)) * (force - sforce[1]);
	TSMSmoothedForce = sforce;
