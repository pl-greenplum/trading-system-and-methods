	inputs:	period(20);
	vars:		ix(0), SK(0), sum(0), avg(0);
	
	avg = average(close,period);
	sum = 0;
	for ix = 0 to period-1 begin
		sum = sum + power(close[ix] - avg,3);
		end;
	
	SK = sum/((period - 1)*power(stddev(close,period),3));
	
	plot1(SK,"Skew");
